import org.apache.camel.builder.AdviceWith;
import org.apache.camel.builder.RouteBuilder;

import static org.apache.camel.builder.Builder.constant;

public class RouteWithExpressions extends RouteBuilder {

    @Override
    public void configure() throws Exception {

        from("timer:foo")
            .setBody(simple("xxx"))
            .setBody().simple("xxx", String.class)
            .setBody().constant("xxx");

        from("disruptor:foo")
            .transform(constant("Bye World"))
            .to("mock:result");

        from("direct:a")
            .split(
                expression()
                    .tokenize()
                    .token("\n")
                .end())
            .to("mock:a");

        from("direct:b")
            .setBody().expression(
                expression()
                    .simple()
                    .expression("Hello World Out")
                .end())
            .to("mock:b");
        from("direct:c")
            .setBody(
                expression()
                    .simple()
                    .expression("Hello World In")
                .end())
            .to("mock:c");
        from("direct:d")
            .filter(
                expression(
                    expression()
                        .header()
                        .expression("foo")
                    .end())
                .isEqualTo("bar"))
            .to("mock:d");

        AdviceWith.adviceWith(context, "advice-with-on-exception-transacted-test-route", a -> {
            a.weaveAddFirst().transform(constant("Bye World"));
        });
    }
}
